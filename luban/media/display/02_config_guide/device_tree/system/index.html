6.3.2.2.1.1. fb0¶

display-fb {
    compatible = "artinchip,aic-framebuffer";
    #address-cells = <1>;
    #size-cells = <0>;
    fb0: fb@0 {
        reg = <0x0 0x0>;
    };
};

6.3.2.2.1.2. display engine¶

de0: de@18a00000 {
    #address-cells = <1>;
    #size-cells = <0>;
    compatible = "artinchip,aic-de-v1.0";
    reg = <0x0 0x18a00000 0x0 0x1000>;
    interrupts-extended = <&plic0 59 IRQ_TYPE_LEVEL_HIGH>;
    clocks = <&cmu CLK_DE>, <&cmu CLK_PIX>;
    clock-names = "de0", "pix";
    resets = <&rst RESET_DE>;
    reset-names = "de0";
    mclk-rate = <200000000>;
};
